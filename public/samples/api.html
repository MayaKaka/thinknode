Ticker

高频计时器

Static
  • Ticker.destroyed = true; // 销毁所有计时器
Constructor
  • new Ticker(fps, useAnimationFrame);
Properties
fps（实际执行帧频）
_fps（初始化设置帧频） _timer（原生计时器） _paused（是否暂停） _targets（执行对象集合）
Methods
  • tikcer.isActive() ==== true; //判断是否在运行
  • tikcer.start(); //启动计时器
  • tikcer.stop(); //关闭计时器
  • ticker.has(func) === true; //判断是否存在执行对象
  • ticker.add(func); ticker.add(obj); //添加执行对象
  • ticker.remove(func); ticker.remove(obj); //移除执行对象
  • _clearTimer（清除原生计时器）_setTimer（创建原生计时器）_exec（执行心跳） 

var ticker = new Ticker(30, 'auto');
ticker.add(function(delta) {
   console.log(delta);	
});
ticker.add({
   count: 0,
   update: (delta){ this.count++; }
});
ticker.start();

StyleSheet

扩展样式

jQuery.extend( StyleSheet.styles, {
	opacity: {
		get: function(target, key) {
			return target.opacity;   
		},
		set: function(target, key, value) {
			target.opacity = value;
			if (!target.renderInCanvas) {
				target.elem.style.opacity = value;
			}
		},
		step: function(target, key, fx) {
			var pos = fx.pos,
				start = fx.start,
				end = fx.end;
			target.style('opacity', (end-start)*pos+start);
		}
	}
});
displayObj.style('opacity', 0.5);
